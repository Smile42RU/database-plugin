CREATE TABLE views_connections (
  id               varchar(50) NOT NULL, 
  version          int(10) NOT NULL, 
  container_id     varchar(50) NOT NULL, 
  view_id          varchar(50) NOT NULL, 
  view_version     int(10) NOT NULL, 
  class            varchar(50) NOT NULL, 
  name             varchar(1024), 
  documentation    tinytext, 
  is_locked        tinyint(1), 
  line_color       varchar(7), 
  line_width       int(10), 
  font             varchar(150), 
  font_color       int(10), 
  relationship_id  varchar(50), 
  source_object_id varchar(50), 
  target_object_id varchar(50), 
  type             int(10), 
  rank             int(10) NOT NULL, 
  checksum         varchar(50) NOT NULL, 
  PRIMARY KEY (id, 
  version, 
  container_id)) comment='views_connections parents can be either a view, a canvas or a sketch but any container as well.';
